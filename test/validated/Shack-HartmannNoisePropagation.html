def wfeVsNoise(magnitude,ron=0,exposure=1e-3):
    src.magnitude = [magnitude]
    nSample = 2500
    cx = np.zeros(nSample)
    cy = np.zeros(nSample)
    for k in range(nSample):
        wfs.reset()
        wfs.propagate(src)
        wfs.readOut(exposure,ron)
        wfs.process()
        c = wfs.c.host()
        cx[k] = c[0,0]
        cy[k] = c[0,1]    
    return src.nPhoton*exposure, cx.var(), cy.var()
